Cross-compile for STM32F4 Discovery board
------------------------------------------
speed
=====
make rebuild TEST=dsp_biquad_df2T_block_f32 TARGET=stm32f4discovery TOOL=armgcc OPT=speed
make rebuild TEST=dsp_biquad_df2T_single_f32 TARGET=stm32f4discovery TOOL=armgcc OPT=speed
make rebuild TEST=dsp_fir_block_f32 TARGET=stm32f4discovery TOOL=armgcc OPT=speed
make rebuild TEST=dsp_fir_single_f32 TARGET=stm32f4discovery TOOL=armgcc OPT=speed
